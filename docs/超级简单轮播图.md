# 超级简单轮播图

 ```javascript
<template>
  <div class="carousel">
    <ul class="carousel-list">
      <li v-for="(item, idx) in images" :key="item.id" v-show="idx === currentIndex" class="carousel-item">
        <img :src="item.url" :alt="item.name" />
      </li>
    </ul>
    <button @click="prev">左</button>
    <button @click="next">右</button>
  </div>
</template>

<script setup>
import { ref, onUnmounted, onMounted } from "vue";

const images = ref([
  { id: 1, name: "Demon 1", url: "https://tse1.mm.bing.net/th/id/OIP.XS6EtBzAmWWa31cfFvcx6QHaHa?w=650&h=651&rs=1&pid=ImgDetMain&o=7&rm=3" },
  { id: 2, name: "Demon 2", url: "https://img.ixintu.com/download/jpg/201911/e25b904bc42a74d7d77aed81e66d772c.jpg!con" },
  { id: 3, name: "Demon 3", url: "https://tse1.mm.bing.net/th/id/OIP.7lURCaoa-V1fPFkaYrhOdwHaHZ?rs=1&pid=ImgDetMain&o=7&rm=3" },
]);

const currentIndex = ref(0);
let timer = null;

const prev = () => {
  currentIndex.value = (currentIndex.value - 1 + images.value.length) % images.value.length;
};
const next = () => {
  currentIndex.value = (currentIndex.value + 1) % images.value.length;
};

const startAuto = () => {
  timer = setInterval(() => {
    next();
  }, 2000);
};

onMounted(() => {
  startAuto();
});
onUnmounted(() => {
  clearInterval(timer);
});

</script>
 ```

```javascript
<template>
  <div>
    <ul>
      <li v-for="(item, idx) in items" :key="item.id" v-show="idx === 0">
        <img :src="item.url" :alt="item.name" style="width:100%;height:100%;object-fit:cover;" />
      </li>
    </ul>
    <button @click="leftFn">左</button>
    <button @click="rightFn">右</button>
  </div>
</template>

<script setup>
import { ref,onUnmounted,onMounted } from "vue";
const items = ref([
  { id: 1, name: "Demon 1" ,url:"https://tse1.mm.bing.net/th/id/OIP.XS6EtBzAmWWa31cfFvcx6QHaHa?w=650&h=651&rs=1&pid=ImgDetMain&o=7&rm=3" },
  { id: 2, name: "Demon 2" ,url:"https://img.ixintu.com/download/jpg/201911/e25b904bc42a74d7d77aed81e66d772c.jpg!con"},
  { id: 3, name: "Demon 3" ,url:"https://tse1.mm.bing.net/th/id/OIP.7lURCaoa-V1fPFkaYrhOdwHaHZ?rs=1&pid=ImgDetMain&o=7&rm=3"},
]);

// 轮播功能
// 先自动轮播
let timer = null;
const leftFn = () => {
  items.value.unshift(items.value.pop());
};
const rightFn = () => {
  items.value.push(items.value.shift());
};
const autoFn = () => {
  timer = setInterval(() => {
    rightFn();
  }, 1000);
};
onMounted(() => {
  autoFn();
});
onUnmounted(() => {
  clearInterval(timer);
});



</script>
```
